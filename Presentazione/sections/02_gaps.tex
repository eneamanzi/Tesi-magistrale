\begin{frame}{Stato dell'arte e gap}
  \begin{itemize}
    \item \textbf{G1 Cecità semantica}: scanner e fuzzer operano sulla \alert{sintassi} HTTP, non sulla \alert{semantica} dei contratti

    \item \textbf{G2 Portabilità/profondità}: gli strumenti specializzati richiedono \alert{riscrittura} per ogni target

    \item \textbf{G3 Riproducibilità}: assenza di \alert{determinismo} preclude l'integrazione in pipeline CI/CD

    \item \textbf{G4 Oracle problem}: nessun meccanismo sistematico per costruire \alert{criteri di valutazione} automatici
  \end{itemize}
\end{frame}

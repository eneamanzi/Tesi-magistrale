\begin{frame}{Stato dell'arte e gap}
  \begin{itemize}
    \item \textbf{G1 Cecità semantica}: scanner e fuzzer operano sulla \alert{sintassi} HTTP, non sulla \alert{semantica} dei contratti

    \item \textbf{G2 Portabilità/profondità}: strumenti specifici sono \alert{approfonditi} ma legati a un \alert{solo contesto}; quelli generici restano \alert{superficiali}

    \item \textbf{G3 Riproducibilità}: assenza di \alert{determinismo} preclude l'integrazione nei cicli di \alert{sviluppo continuativo}

    \item \textbf{G4 Oracle problem}: nessun meccanismo sistematico per stabilire se un comportamento viola una \alert{garanzia di sicurezza}
  \end{itemize}
\end{frame}

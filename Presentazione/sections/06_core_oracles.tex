\begin{frame}{Specified oracles: il criterio di valutazione}
  \begin{itemize}
    \item \textbf{Problema dell'oracolo}: nessun criterio sistematico per giudicare se un \alert{comportamento osservato} è una \alert{violazione}

    \item \textbf{Specified oracles}: criterio definito \textit{a priori} dalla \alert{conoscenza del dominio} del test; garantisce \alert{riproducibilità}

    \item \alert{Fonti eterogenee}, dipendenti dal controllo: \textit{es.} schema OAS, standard tecnici, o costruzioni ad-hoc
  \end{itemize}
\end{frame}

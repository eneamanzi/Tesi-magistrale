\begin{frame}{Specified oracles: il criterio di valutazione}
  \begin{center}
    \small\textit{Come si stabilisce se un comportamento osservato viola una garanzia di sicurezza?}
  \end{center}
  \vspace{0.4em}
  \begin{itemize}
    \item Generare input è risolto; \alert{valutare il comportamento osservato} resta il collo di bottiglia del testing automatizzato

    \item APIGuard adotta \alert{specified oracles}: criteri codificati \alert{a priori} in ogni test, dalla conoscenza del dominio del controllo

    \item Tre fonti: schema \alert{OAS}, standard tecnici, costruzione ad-hoc sul controllo specifico
  \end{itemize}
\end{frame}

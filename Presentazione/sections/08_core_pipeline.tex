\begin{frame}{Pipeline di esecuzione e connector}
  \begin{itemize}
    \item Pipeline a \alert{7 fasi sequenziali}: prime 4 bloccanti (errore interrompe l'avvio), ultime 3 non bloccanti (errori isolati, pipeline prosegue)

    \item Connector \alert{gerarchici} a 3 livelli: separano la raccolta dei dati grezzi dalla valutazione dell'esito; lo stesso connector funziona in locale, CI/CD e ambienti containerizzati

    \item Discovery \alert{dinamico}: nuove verifiche vengono registrate automaticamente senza modifiche al core
  \end{itemize}
\end{frame}

\begin{frame}{Gaps}
  \begin{itemize}
    \item \textbf{G1 Cecità semantica}: i \alert{tool di assessment} operano sulla \alert{sintassi}, non sulla \alert{semantica} dei contratti

    \item \textbf{G2 Portabilità e profondità}: o \alert{specifico} e \textit{approfondito su un target}, o \alert{superficiale} e \textit{portabile ovunque}

    \item \textbf{G3 Riproducibilità}: l'assenza di \alert{determinismo} preclude l'integrazione nei cicli di \alert{sviluppo continuativo}

    \item \textbf{G4 Oracle problem}: carenza di meccanismi sistematici per stabilire se un \alert{comportamento} viola una \alert{garanzia di sicurezza}
  \end{itemize}
\end{frame}

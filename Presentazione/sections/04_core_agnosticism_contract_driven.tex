\begin{frame}{Agnosticismo applicativo e contract-driven}
  \begin{itemize}
    \item \textbf{Agnosticismo applicativo}: uniche sorgenti di conoscenza a runtime, \alert{\texttt{openapi.yaml}} e \alert{\texttt{config.yaml}}
          \begin{itemize}
            \item nessun riferimento hardcoded al target
            \item deployabile su \alert{qualsiasi API REST documentata} senza modifiche
            \item esteso anche al \alert{gateway}: interfaccia astratta indipendente dal prodotto
          \end{itemize}

    \item \textbf{Paradigma contract-driven}: \alert{OAS} guida la \alert{costruzione} delle richieste, delimita la \alert{superficie d'attacco}, funge da \alert{oracolo} per alcune valutazioni
  \end{itemize}
\end{frame}

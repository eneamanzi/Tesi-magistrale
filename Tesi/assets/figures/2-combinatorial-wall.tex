      \node[vis]  (b4) at (a4 |- b1) {Logica\\applicativa\\{\scriptsize [visibile]}};

      \draw[arrA] (a1) -- (a2);

      \draw[gray!40, dashed, thick] ([xshift=-0.5em]lblA.west |- mid-y) -- ([xshift=0.5em]a4.east |- mid-y);
